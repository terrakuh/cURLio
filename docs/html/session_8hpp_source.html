90inline Response Session::start(Request& request)
91{
92 if (!request.is_valid() || request.is_active()) {
93 throw boost::system::system_error{ Code::request_in_use };
94 }
95
96 CURLIO_DEBUG("Starting request " << &request);
97 auto data = std::move(request._owner);
98 const auto handle = data->handle;
99 data->executor = get_executor();
100
101 curl_easy_setopt(handle, CURLOPT_OPENSOCKETFUNCTION, &Session::_open_socket);
102 curl_easy_setopt(handle, CURLOPT_OPENSOCKETDATA, this);
103 curl_easy_setopt(handle, CURLOPT_CLOSESOCKETFUNCTION, &Session::_close_socket);
104 curl_easy_setopt(handle, CURLOPT_CLOSESOCKETDATA, this);
105 curl_easy_setopt(handle, CURLOPT_PRIVATE, data.get());
106 curl_easy_setopt(handle, CURLOPT_COOKIEFILE, _cookie_file.c_str());
107 if (!_cookie_file.empty()) {
108 CURLIO_DEBUG("Setting cookie file '" << _cookie_file << "' for request " << &request);
109 curl_easy_setopt(handle, CURLOPT_COOKIEJAR, _cookie_file.c_str());
110 }
111 curl_easy_setopt(handle, CURLOPT_SHARE, _share_handle);
112
113 Response response{ data };
114 _active_requests.insert({ data.get(), std::move(data) });
115
116 // kickstart
117 boost::asio::post(_strand, [this, handle] {
118 curl_multi_add_handle(_multi_handle, handle);
119 _multi_timer_callback(_multi_handle, 0, this);
120 });
121
122 return response;
123}

125inline void Session::_clean_finished() noexcept
126{
127 CURLIO_TRACE("Cleaning finished requests");
128 CURLMsg* message = nullptr;
129 int left = 0;
130 while ((message = curl_multi_info_read(_multi_handle, &left))) {
131 if (message->msg == CURLMSG_DONE) {
132 detail::Shared_data* data = nullptr;
133 curl_multi_remove_handle(_multi_handle, message->easy_handle);
134 if (curl_easy_getinfo(message->easy_handle, CURLINFO_PRIVATE, &data) == CURLE_OK && data != nullptr) {
135 CURLIO_DEBUG("Request " << data << " is done");
136 _active_requests.erase(data);
137 }
138 }
139 }

142inline void Session::_async_wait(boost::asio::ip::tcp::socket& socket,
143 boost::asio::socket_base::wait_type type)
144{
145 const auto handle = socket.native_handle();
146 socket.async_wait(type, [this, type, handle](boost::system::error_code ec) {
147 CURLIO_TRACE("Socket action=" << (type == boost::asio::socket_base::wait_read ? "READ" : "WRITE")
148 << " ec=" << ec.what());
149 const int mask = (type == boost::asio::socket_base::wait_read ? CURL_CSELECT_IN : CURL_CSELECT_OUT) |
150 (ec ? CURL_CSELECT_ERR : 0);
151 int still_running = 0;
152 const auto code = curl_multi_socket_action(_multi_handle, handle, mask, &still_running);
153 if (code != CURLM_OK) {
154 CURLIO_ERROR("Socket action: " << curl_multi_strerror(code));
155 }
156 _clean_finished();
157
158 const auto it = _sockets.find(handle);
159 // TODO can multiple same wait operations
160 // TODO support write cycle
161 if (!ec && it != _sockets.end() && type == boost::asio::socket_base::wait_read && it->second.watch_read) {
162 _async_wait(it->second.socket, type);
163 }
164 });
165}

167inline int Session::_socket_callback(CURL* handle, curl_socket_t socket, int what, void* self_pointer,
168 void* socket_pointer) noexcept
169{
170 constexpr const char* what_names[] = { "IN", "OUT", "IN/OUT", "REMOVE" };
171 CURLIO_TRACE(
172 "Socket action callback; action=" << (what >= 1 && what <= 4 ? what_names[what - 1] : "unknwon"));
173 const auto self = static_cast<Session*>(self_pointer);
174 const auto it = self->_sockets.find(socket);
175 if (it == self->_sockets.end()) {
176 return 0;
177 }
178
179 if (what & CURL_POLL_IN) {
180 it->second.watch_read = true;
181 }
182 if (what & CURL_POLL_OUT) {
183 it->second.watch_write = true;
184 }
185
186 switch (what) {
187 case CURL_POLL_IN: self->_async_wait(it->second.socket, boost::asio::socket_base::wait_read); break;
188 case CURL_POLL_OUT: self->_async_wait(it->second.socket, boost::asio::socket_base::wait_write); break;
189 case CURL_POLL_INOUT: {
190 self->_async_wait(it->second.socket, boost::asio::socket_base::wait_read);
191 self->_async_wait(it->second.socket, boost::asio::socket_base::wait_write);
192 break;
193 }
194 case CURL_POLL_REMOVE: {
195 it->second.watch_read = false;
196 it->second.watch_write = false;
197 it->second.socket.cancel();
198 break;
199 }
200 }
201 return 0;
202}

204inline int Session::_multi_timer_callback(CURLM* multi, long timeout_ms, void* self_pointer) noexcept
205{
206 CURLIO_DEBUG("Starting timeout with " << timeout_ms << " ms");
207 const auto self = static_cast<Session*>(self_pointer);
208
209 self->_timer.expires_after(std::chrono::milliseconds{ timeout_ms });
210 self->_timer.async_wait([self](boost::system::error_code ec) {
211 CURLIO_TRACE("Timeout with ec=" << ec.what());
212 if (!ec) {
213 int still_running = 0;
214 const auto code = curl_multi_socket_action(self->_multi_handle, CURL_SOCKET_TIMEOUT, 0, &still_running);
215 if (code == CURLM_OK) {
216 CURLIO_DEBUG(still_running << " requests still running");
217 self->_clean_finished();
218 } else {
219 CURLIO_ERROR("====================" << curl_multi_strerror(code));
220 }
221 }
222 });
223 CURLIO_DEBUG("Exiting multi timer");
224
225 return 0;
226}

228inline curl_socket_t Session::_open_socket(void* self_pointer, curlsocktype purpose,
229 struct curl_sockaddr* address) noexcept
230{
231 CURLIO_TRACE("Trying to open socket");
232 const auto self = static_cast<Session*>(self_pointer);
233 if (purpose == CURLSOCKTYPE_IPCXN) {
234 if (address->family == AF_INET) {
235 boost::system::error_code ec;
236 boost::asio::ip::tcp::socket socket{ self->get_executor() };
237 socket.open(boost::asio::ip::tcp::v4(), ec);
238 if (!ec) {
239 auto fd = socket.native_handle();
240 CURLIO_INFO("New socket opened: " << fd);
241 self->_sockets.insert(std::make_pair(fd, std::move(socket)));
242 return fd;
243 }
244 }
245 // TODO add IPv6 support
246 }
247 CURLIO_ERROR("Failed to open socket. Purpose=" << purpose << ", family=" << address->family);
248 return CURL_SOCKET_BAD;
249}
